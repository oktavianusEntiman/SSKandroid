141      @NonNull 
142      public static ArrayRealVector readVectorFromFile(@NonNull String fileName) throws FileNotFoundException 
143      { 
144          File file =new File(fileName); 
145          Scanner sc = new Scanner(file); 
146          String line; 
147          String[] tokens; 
148          ArrayRealVector vec = new ArrayRealVector(); 
149          while (sc.hasNextLine()) { 
150              line  = sc.nextLine(); 
151              tokens = line.split(","); 
152              for(int i=0;i<tokens.length;i++) 
153                  vec = (ArrayRealVector) vec.append(Double.parseDouble(tokens[i])); 
154          } 
155          sc.close(); 
156          return vec; 
157      }

159      @Nullable 
160      public static RealMatrix readMatrixFromFile(@NonNull String fileName) 
161      { 
162          RealMatrix mat = null ; 
163          try { 
164              File file =new File(fileName); 
165              Scanner sc = new Scanner(file); 
166              String line,rows,columns; 
167              String[] tokens; 
168              double[][] data = null; 
169              int i=0; 
170              rows=sc.nextLine(); 
171              columns=sc.nextLine(); 
172              logMessage(TAG, rows+columns); 
173              data = new double[Integer.parseInt(rows)][Integer.parseInt(columns)]; 
174              while (sc.hasNextLine()) { 
175                  line  = sc.nextLine(); 
176                  logMessage(TAG, line); 
177                  tokens = line.split(","); 
178                  for(int j=0;j<Integer.parseInt(columns);j++) 
179                  { 
180                      data[i][j] = Double.parseDouble(tokens[j]); 
181                  } 
182                  i++; 
183              } 
184              sc.close(); 
185              mat = MatrixUtils.createRealMatrix(data); 
186              return mat;} 
187   
188          catch(Exception e) 
189          { 
190              logMessage(TAG, e.getMessage()); 
191              return mat; 
192          }

295          String line;

297          try {

307              else {

318          }

346      public static dbModule retrieveDefaultModule (@NonNull Context context, @Nullable String moduleName /*Set this var with null for default module*/){ 
347          if (moduleName == null){ 
348              moduleName = mySettings.getPrimaryModuleName(); 
349          } 
350          dbModule tmpModule = new dbModule(moduleName); 
351          JSONParser parser = new JSONParser(); 
352          JSONObject moduleObj; 
353          try { 
354   
355              moduleObj = (JSONObject) parser.parse(readFileAsString(context, moduleName + ".json", true)); 
356              if (moduleObj == null){ 
357                  logMessage(TAG, "Json Module Obj is null"); 
358                  return null; 
359              } 
360              logMessage(TAG, "------- Reading module data ... --------"); 
361              moduleName = (String) moduleObj.get("moduleName"); 
362   
363  //            String moduleInfoFileName = (String) moduleObj.get("moduleInfo"); 
364  //            logMessage(TAG, "Module info file Name: " + moduleInfoFileName); 
365   
366              JSONObject jsonModulesInfo = (JSONObject) parser.parse(readFileAsString(context, "ModulesInfo.json", true)); 
367   
368              JSONObject jsonCurrentModuleInfo = (JSONObject) jsonModulesInfo.get(moduleName); 
369   
370              String tmp = jsonModulesInfo.keySet().toString(); 
371              String[] keys = tmp.substring(1, tmp.length()-1).split(", "); 
372   
373   
374              // Create list of my modules 
375              dbModuleInfo moduleInfo; 
376              ArrayList<dbModuleInfo> myModuleees = new ArrayList<>(); 
377              for (String modl : keys) { 
378                  logMessage(TAG, "Var name: " + keys); 
379                  JSONObject obj = (JSONObject) jsonModulesInfo.get(modl); 
380                  if (obj == null) logMessage(TAG, "NULLLLLLLLLLLLLL"); 
381                  moduleInfo = getModuleInfo(context, obj); 
382                  logMessage(TAG, "Variable: " + obj.toString()); 
383                  myModuleees.add(moduleInfo); 
384                  logMessage(TAG, "New my modules list equals: " + myModuleees.size()); 
385              } 
386              myModules = myModuleees; 
387   
388              if (jsonCurrentModuleInfo == null) logMessage(TAG, "json Module Info is nuuuuulllllll"); 
389              if (!jsonCurrentModuleInfo.get("name").toString().toLowerCase().equals(moduleName.toLowerCase())){ 
390  //                Toast.makeText(context, "Error in reading module info.", Toast.LENGTH_SHORT).show(); 
391                  logMessage(TAG, "Error in reading module info."); 
392                  return null; 
393              } 
394   
395              String bgFileName = (String) moduleObj.get("bgFile"); 
396              Bitmap moduleBG = ImageUtil.convert(readFileAsString(context, bgFileName, true)); 
397              tmpModule.setModuleBackground(moduleBG); 
398   
399              JSONObject jsonVariables = (JSONObject) moduleObj.get("variables"); 
400   
401              logMessage(TAG, "Module Name: " + moduleName); 
402              logMessage(TAG, "background file Name: " + bgFileName); 
403   
404              tmp = jsonVariables.keySet().toString(); 
405              String[] varNames = tmp.substring(1, tmp.length()-1).split(", "); 
406              logMessage(TAG, "Variables: " + Arrays.toString(varNames)); 
407   
408              dbVariable var; 
409              ArrayList<dbVariable> variables = new ArrayList<>(); 
410              for (String varName : varNames) { 
411                  logMessage(TAG, "Var name: " + varName); 
412                  var = new dbVariable(varName); 
413                  JSONObject obj = (JSONObject) jsonVariables.get(varName); 
414                  if (obj == null) logMessage(TAG, "NULLLLLLLLLLLLLL"); 
415                  var = readVariable(var, obj); 
416                  logMessage(TAG, "Variable: " + var.toString()); 
417                  variables.add(var); 
418                  logMessage(TAG, "New length of variables list equals: " + variables.size()); 
419              } 
420              tmpModule.setModuleVariables(variables); 
421   
422              // Reading module info 
423              dbModuleInfo info = getModuleInfo(context, jsonCurrentModuleInfo); 
424   
425              logMessage(TAG, "Module Info: " + info.toString()); 
426   
427              tmpModule.setModuleInfo(info); 
428   
429              currentModule = tmpModule; 
430              logMessage(TAG, "Variables in current module: " + currentModule.getModuleVariables().size()); 
431  //        return tmpModule; 
432          } catch (ParseException e) { 
433              logMessage(TAG, e.getLocalizedMessage()); 
434          } catch (Exception e) { 
435              logMessage(TAG, e.getLocalizedMessage()); 
436          } 
437   
438          return tmpModule;
