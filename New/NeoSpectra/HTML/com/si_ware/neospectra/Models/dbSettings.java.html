41           ourInstance.primaryModuleName = "Milk"; 
42           ourInstance.useDefaultStorageSize = true; 
43           ourInstance.storageSize = defaultStorageSize; // Megabyte. 
44           ourInstance.defaultCaptureState = captureImage.Ask_Every_time;

48       @NonNull 
49       public dbSettings getCurrentUserSettings(@NonNull Context mContext){ 
50           SharedPreferences sharedPref = mContext.getSharedPreferences( 
51                   mContext.getResources().getString(R.string.key_Preferences_File), 
52                   Context.MODE_PRIVATE); 
53           // read the data from SharedPreferences and set the default values if it not found. 
54           ourInstance.primaryModuleName = sharedPref.getString("primaryModuleName", "Milk"); 
55           ourInstance.defaultCaptureState = captureImage.valueOf(sharedPref.getString("captureState", 
56                   captureImage.Ask_Every_time.toString())); 
57           ourInstance.storageSize = sharedPref.getInt("storageSize", defaultStorageSize); 
58           ourInstance.useDefaultStorageSize = sharedPref.getBoolean("isDefaultStorageSize", true); 
59           return ourInstance; 
60       } 
61    
62       public void saveUserSettings(@NonNull Context mContext){ 
63           //Save user setting to shared preferences 
64           SharedPreferences sharedPref = mContext.getSharedPreferences( 
65                   mContext.getResources().getString(R.string.key_Preferences_File), 
66                   Context.MODE_PRIVATE); 
67           SharedPreferences.Editor editor = sharedPref.edit(); 
68           editor.putString("primaryModuleName", primaryModuleName); 
69           editor.putString("captureState", defaultCaptureState.toString()); 
70           editor.putInt("storageSize", storageSize); 
71           editor.putBoolean("isDefaultStorageSize", useDefaultStorageSize); 
72           editor.apply(); 
73       }
