286      private BitmapDrawable writeOnDrawable(int drawableId, @NonNull String text){ 
287   
288          BitmapFactory.Options options = new BitmapFactory.Options(); 
289          options.inScaled = false; 
290  //        Bitmap bm = BitmapFactory.decodeResource(getResources(), drawableId, options).copy(Bitmap.Config.ARGB_8888, true); 
291   
292          Bitmap bmp1 = BitmapFactory.decodeResource(this.getResources(), drawableId); 
293          //then create a copy of bitmap bmp1 into bmp2 
294          Bitmap bm = bmp1.copy(bmp1.getConfig(), true); 
295   
296          if(paint == null){ 
297              paint = new Paint(Paint.ANTI_ALIAS_FLAG); 
298          } 
299          paint.setStyle(Paint.Style.FILL); 
300          paint.setColor(getResources().getColor(R.color.grey)); //Change this if you want other color of text 
301          paint.setTextSize(30); //Change this if you want bigger/smaller font 
302   
303          Canvas canvas = new Canvas(bm); 
304          canvas.drawText(text, 10, bm.getHeight()/4, paint); //Change the position of the text here 
305   
306          return new BitmapDrawable(getResources(), bm); 
307      }

550      private void firebaseAuthWithGoogle(GoogleSignInAccount acct) { 
551          Log.d(TAG, "firebaseAuthWithGoogle:" + acct.getId()); 
552   
553          AuthCredential credential = GoogleAuthProvider.getCredential(acct.getIdToken(), null); 
554          mAuth.signInWithCredential(credential) 
555                  .addOnCompleteListener(this, new OnCompleteListener<AuthResult>() { 
556                      @Override 
557                      public void onComplete(@NonNull Task<AuthResult> task) { 
558                          if (task.isSuccessful()) { 
559                              // Sign in success, update UI with the signed-in user's information 
560                              Log.d(TAG, "signInWithCredential:success"); 
561                              FirebaseUser user = mAuth.getCurrentUser(); 
562                              updateUI(user); 
563                          } else { 
564                              // If sign in fails, display a message to the user. 
565                              Log.w(TAG, "signInWithCredential:failure", task.getException()); 
566                              Snackbar.make(findViewById(R.id.main_layout), "Authentication Failed.", Snackbar.LENGTH_SHORT).show(); 
567                              updateUI(null);
