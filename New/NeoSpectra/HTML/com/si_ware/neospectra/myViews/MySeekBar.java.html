44       @Override 
45       public synchronized void setProgress(int progress) { 
46           super.setProgress(progress); 
47       } 
48    
49       public void initData(ArrayList<ProgressItem> progressItemsList) { 
50           this.mProgressItemsList = progressItemsList; 
51       } 
52    
53       @NonNull 
54       public Drawable getThumb(int progress) { 
55           ((TextView) thumbView.findViewById(R.id.tv_seekbar_value)).setText(progress + ""); 
56           thumbView.measure(View.MeasureSpec.UNSPECIFIED, View.MeasureSpec.UNSPECIFIED); 
57           Bitmap bitmap = Bitmap.createBitmap(thumbView.getMeasuredWidth(), thumbView.getMeasuredHeight(), Bitmap.Config.ARGB_8888); 
58           Canvas canvas = new Canvas(bitmap); 
59           thumbView.layout(0, 0, thumbView.getMeasuredWidth(), thumbView.getMeasuredHeight()); 
60           thumbView.draw(canvas); 
61           return new BitmapDrawable(getResources(), bitmap); 
62       } 
63    
64       @Override 
65       protected synchronized void onDraw(@NonNull Canvas canvas) { 
66           int progressBarWidth = getWidth(); 
67           int progressBarHeight = getHeight(); 
68           int thumboffset = getThumbOffset(); 
69   //        setThumbOffset(20); 
70           int lastProgressX = 0; 
71           int progressItemWidth, progressItemRight; 
72           for (int i = 0; i < mProgressItemsList.size(); i++) { 
73               ProgressItem progressItem = mProgressItemsList.get(i); 
74               Paint progressPaint = new Paint(); 
75               progressPaint.setColor(getResources().getColor(progressItem.color)); 
76               progressItemWidth = (int) (progressItem.progressItemPercentage 
77                       * progressBarWidth / 100); 
78               progressItemRight = lastProgressX + progressItemWidth; 
79               // for last item give right of the progress item to width of the 
80               // progress bar 
81               if (i == mProgressItemsList.size() - 1 
82                       && progressItemRight != progressBarWidth) { 
83                   progressItemRight = progressBarWidth; 
84               } 
85               Rect progressRect = new Rect(); 
86               progressRect.set(lastProgressX, thumboffset / 2, progressItemRight, 
87                       progressBarHeight - thumboffset / 2); 
88               canvas.drawRect(progressRect, progressPaint); 
89               lastProgressX = progressItemRight; 
90           } 
91           super.onDraw(canvas); 
92       }
