3    import android.view.animation.Animation; 
4    import android.view.animation.Transformation; 
5    import android.widget.ProgressBar; 
6     
7    /** 
8     * Created by AmrWinter on 12/17/2017.

10    
11   public class ProgressBarAnimation extends Animation { 
12       private ProgressBar progressBar; 
13       private float from; 
14       private float  to; 
15    
16       public ProgressBarAnimation(ProgressBar progressBar, float from, float to) { 
17           super(); 
18           this.progressBar = progressBar; 
19           this.from = from; 
20           this.to = to; 
21       } 
22    
23       @Override 
24       protected void applyTransformation(float interpolatedTime, Transformation t) { 
25           super.applyTransformation(interpolatedTime, t); 
26           float value = from + (to - from) * interpolatedTime; 
27           progressBar.setProgress((int) value); 
28       } 
29    
30   }
