19   public class RunModule { 
20    
21       private String TAG = getClass().getSimpleName(); 
22       private IRegressionData mVarData; 
23       private dbVariable mVariable; 
24       private IModule mModule; 
25       public RunModule(dbVariable pVariable) { 
26           mVariable = pVariable; 
27           mVarData = mVariable.getVariableData().getRegressionData(); 
28       } 
29    
30       @Nullable 
31       public dbVarResult run(ArrayRealVector lastProcessResult) { 
32           dbVarResult tempResult; 
33           Double result = 0.0; 
34           GlobalVariables.regression type = mVarData.getType(); 
35           try { 
36                switch (type){ 
37                    case GPR: 
38                        mModule = new GPR(mVarData); 
39                        result = mModule.getResult(lastProcessResult); 
40                        break; 
41                    case PCR: 
42                        mModule = new PCR(mVarData); 
43                        result = mModule.getResult(lastProcessResult); 
44                        break; 
45                    case SVR: 
46                        mModule = new SVR(mVarData); 
47                        result = mModule.getResult(lastProcessResult); 
48                        break; 
49                    case PLS: 
50                        mModule = new PLS(mVarData, false); 
51                        result = mModule.getResult(lastProcessResult); 
52                        break; 
53                    default: 
54                        logMessage(TAG, "Wrong regression module type"); 
55                } 
56                tempResult = new dbVarResult(mVariable.getVarName(), mVariable.getResultType(), result); 
57           }catch (Exception e){ 
58               logMessage(TAG, "Error, RunModule may be null"); 
59               tempResult = null; 
60           } 
61           return tempResult; 
62       } 
63   }
