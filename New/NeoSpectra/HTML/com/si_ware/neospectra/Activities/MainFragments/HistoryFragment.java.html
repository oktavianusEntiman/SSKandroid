84       @NonNull 
85       private ArrayList<dbResult> Filter(ArrayList<dbResult> list, @NonNull String query){ 
86           ArrayList<dbResult> newList = new ArrayList<>(); 
87           for (dbResult result : list) { 
88               if (result.getModuleName().toLowerCase().startsWith(query.toLowerCase()) 
89                       || result.getTimestamp().contains(query)){ 
90                   newList.add(result); 
91               } 
92           } 
93   //        Log.v(TAG, "newList size" + newList.size()); 
94           return newList; 
95       }

103              dialog.setTitle("No results"); 
104              dialog.show(); 
105          } 
106      }
