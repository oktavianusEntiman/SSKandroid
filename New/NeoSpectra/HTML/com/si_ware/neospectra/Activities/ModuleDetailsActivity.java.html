54       private void loadModuleInfo(@NonNull dbModuleInfo info) { 
55           try { 
56               // To avoid any null pointer exception right here. 
57               description.setText(info.getDescription()); 
58               moduleIcon.setImageBitmap(info.getIcon()); 
59               moduleName.setText(info.getName()); 
60   //            headerImage.setBackground(new BitmapDrawable(info.getBackground())); 
61               isPaidModule = info.isPaidModule(); 
62               if (isPaidModule){ 
63                   price.setVisibility(View.GONE); 
64                   buyUse.setText(getResources().getString(R.string.use)); 
65               }else { 
66                   price.setText(info.getPrice() + "$"); 
67               } 
68           }catch (Exception e){ 
69               logMessage(TAG, e.getMessage()); 
70           } 
71    
72       }

87       @Override 
88       public void onBackPressed() { 
89           super.onBackPressed(); 
90           logMessage(TAG, "From Equals: " + from); 
91           if (from == null){ 
92               gotoScan(); 
93               return; 
94           } 
95           switch (from){ 
96               case "scan": 
97                   gotoScan(); 
98                   break; 
99               case "buy": 
100                  gotoBuy(); 
101                  break; 
102              case "history": 
103                  gotoHistory(); 
104                  break; 
105              default: 
106                  gotoScan(); 
107          } 
108      }
