16   package com.si_ware.neospectra.InAppBillingUtil; 
17    
18   import android.support.annotation.NonNull; 
19    
20   /** 
21    * Exception thrown when something went wrong with in-app billing. 
22    * An IabException has an associated IabResult (an error). 
23    * To get the IAB result that caused this exception to be thrown, 
24    * call {@link #getResult()}. 
25    */ 
26   public class IabException extends Exception { 
27       IabResult mResult; 
28    
29       public IabException(@NonNull IabResult r) { 
30           this(r, null); 
31       } 
32    
33       public IabException(int response, String message) { 
34           this(new IabResult(response, message)); 
35       } 
36    
37       public IabException(IabResult r, Exception cause) { 
38           super(r.getMessage(), cause); 
39           mResult = r; 
40       } 
41    
42       public IabException(int response, String message, Exception cause) { 
43           this(new IabResult(response, message), cause); 
44       } 
45    
46       /** Returns the IAB result (error) that this exception signals. */ 
47       public IabResult getResult() { return mResult; } 
48   }
