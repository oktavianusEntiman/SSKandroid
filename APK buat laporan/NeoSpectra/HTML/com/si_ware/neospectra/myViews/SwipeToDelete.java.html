67       private void init() { 
68           background = new ColorDrawable(); 
69           xMarkMargin = (int) context.getResources().getDimension(R.dimen.txtMargin); 
70           deleteIcon = ContextCompat.getDrawable(context, android.R.drawable.ic_menu_delete); 
71           deleteIcon.setColorFilter(Color.WHITE, PorterDuff.Mode.SRC_ATOP); 
72           initiated = true; 
73       }

89       @Override 
90       public void onChildDraw(@NonNull Canvas c, RecyclerView recyclerView, RecyclerView.ViewHolder viewHolder, 
91                               float dX, float dY, int actionState, boolean isCurrentlyActive) { 
92    
93           View itemView = viewHolder.itemView; 
94           if (!initiated) { 
95               init(); 
96           } 
97    
98           int itemHeight = itemView.getBottom() - itemView.getTop(); 
99           //Setting Swipe Background 
100          ((ColorDrawable) background).setColor(getLeftColorCode()); 
101          background.setBounds(itemView.getRight() + (int) dX, itemView.getTop(), 
102                  itemView.getRight(), itemView.getBottom()); 
103          background.draw(c); 
104   
105          int intrinsicWidth = deleteIcon.getIntrinsicWidth(); 
106          int intrinsicHeight = deleteIcon.getIntrinsicWidth(); 
107   
108          int xMarkLeft = itemView.getRight() - xMarkMargin - intrinsicWidth; 
109          int xMarkRight = itemView.getRight() - xMarkMargin; 
110          int xMarkTop = itemView.getTop() + (itemHeight - intrinsicHeight) / 2; 
111          int xMarkBottom = xMarkTop + intrinsicHeight; 
112   
113          //Setting Swipe Icon 
114          deleteIcon.setBounds(xMarkLeft, xMarkTop + 16, xMarkRight, xMarkBottom); 
115          deleteIcon.draw(c); 
116   
117          //Setting Swipe Text 
118          Paint paint = new Paint(); 
119          paint.setColor(Color.WHITE); 
120          paint.setTextSize(48); 
121          paint.setTextAlign(Paint.Align.CENTER); 
122          c.drawText(getLeftSwipeLabel(), xMarkLeft + 10, xMarkTop + 10, paint); 
123   
124          super.onChildDraw(c, recyclerView, viewHolder, dX, dY, actionState, isCurrentlyActive); 
125      }
