134      private void ReplaceFragment(int fragment_id) { 
135          myFragment = null; 
136          switch (fragment_id){ 
137              case R.id.scan: 
138                  myFragment = new ScanFragment(); 
139                  break; 
140              case R.id.iv_change_module: 
141                  myFragment = new MyModulesFragment(); 
142                  break; 
143              case R.id.history: 
144                  myFragment = new HistoryFragment(); 
145                  break; 
146              case R.id.shop: 
147                  myFragment = new BuyModulesFragment(); 
148                  break; 
149              case R.id.nav_about_us: 
150                  Log.v("btnPress", "About Us"); 
151                  myFragment = new AboutFragment(); 
152                  break; 
153          } 
154          FragmentManager fragmentManager = getSupportFragmentManager(); 
155          try{ 
156              String FragmentName = myFragment.getClass().getName(); 
157              fragmentManager.beginTransaction() 
158                      .replace(R.id.fragment_main, myFragment) 
159                      .addToBackStack(FragmentName).commit(); 
160          }catch (NullPointerException e){ 
161              Log.e("err", e.getMessage() + ""); 
162          } 
163      }

165      public void onClick(@NonNull View view) { 
166          Toast.makeText(this, "Clicked on button", Toast.LENGTH_SHORT).show(); 
167  //        view.setLayoutParams(new ConstraintLayout.LayoutParams(70, 90)); 
168          // TODO: 4/19/18 Animate scale here 
169          int id = view.getId(); 
170          switch (id){ 
171              case R.id.shop: 
172                  displayFragment(GOTO_FRAGMENT_SHOP); 
173                  break; 
174              case R.id.history: 
175                  displayFragment(GOTO_FRAGMENT_HISTORY); 
176                  break; 
177              default:{ 
178                  logMessage(TAG, "Wrong button id"); 
179                  displayFragment(GOTO_FRAGMENT_SCAN); 
180              } 
181          } 
182      }

184      private void displayFragment(int GOTO){ 
185          int id; 
186          switch (GOTO){ 
187              case GOTO_FRAGMENT_SHOP: 
188                  id = R.id.shop; 
189                  break; 
190              case GOTO_FRAGMENT_ABOUT_US: 
191                  id = R.id.nav_about_us; 
192                  break; 
193              case GOTO_FRAGMENT_HISTORY: 
194                  id = R.id.history; 
195                  break; 
196              default:{ 
197                  logMessage(TAG, "Wrong button id"); 
198                  id = R.id.scan; 
199              } 
200          } 
201          ReplaceFragment(id); 
202          activateView(id); 
203      }

205      private void activateView(int viewID){ 
206          for (int i: bottom_buttons_IDs){ 
207              View v = findViewById(i); 
208              TextView tv = v.findViewWithTag(getResources().getString(R.string.bb_tv_tag)); 
209              ImageView iv = v.findViewWithTag(getResources().getString(R.string.bb_iv_tag)); 
210              int ivID = iv.getId(); 
211              if (viewID == v.getId()){ 
212                  tv.setTextColor(getResources().getColor(R.color.colorPrimary)); 
213                  switch (ivID){ 
214                      case R.id.iv_shop: 
215                          activeViw = "shop"; 
216                          iv.setImageDrawable(getResources().getDrawable(R.drawable.shop_active)); 
217                          break; 
218                      case R.id.iv_scan: 
219                          activeViw = "scan"; 
220                          iv.setImageDrawable(getResources().getDrawable(R.drawable.scan_active_icon)); 
221                          break; 
222                      case R.id.iv_history: 
223                          activeViw = "history"; 
224                          iv.setImageDrawable(getResources().getDrawable(R.drawable.history_active)); 
225                          break; 
226                  } 
227              }else { 
228                  tv.setTextColor(getResources().getColor(R.color.grey)); 
229                  switch (ivID){ 
230                      case R.id.iv_shop: 
231                          iv.setImageDrawable(getResources().getDrawable(R.drawable.shop_icon)); 
232                          break; 
233                      case R.id.iv_scan: 
234                          iv.setImageDrawable(getResources().getDrawable(R.drawable.scan_icon)); 
235                          break; 
236                      case R.id.iv_history: 
237                          iv.setImageDrawable(getResources().getDrawable(R.drawable.history_icon)); 
238                          break; 
239                  } 
240              } 
241   
242          } 
243      }
