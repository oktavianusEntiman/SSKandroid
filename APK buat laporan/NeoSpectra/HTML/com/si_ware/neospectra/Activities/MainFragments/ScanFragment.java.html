172              try { 
173                  doScan(); 
174              } catch (Exception e) { 
175                  e.printStackTrace(); 
176              } 
177          }

284      @NonNull 
285      private BroadcastReceiver mMessageReceiver = new BroadcastReceiver() { 
286          @Override 
287          public void onReceive(Context context, Intent intent) { 
288              markProcessingComplete(); 
289              String intentName = intent.getStringExtra("iName"); 
290              switch (intentName){ 
291                  case "sensorNotification": 
292                      if (intent.getBooleanExtra("isNotificationSuccess", false)){ 
293                          Toast.makeText(context, "Error in sensor reading", 
294                                  Toast.LENGTH_SHORT).show(); 
295                          // Log the Error message 
296                          logMessage("bluetooth", intent.getStringExtra("err")); 
297                      }else { 
298                          // Here we got a successful reading from the sensor 
299                          // pass that reading to the ML code and do analysis 
300                          double[] reading = intent.getDoubleArrayExtra("data"); 
301                          logMessage("bluetooth", "The sensor reading: " + Arrays.toString(reading)); 
302                          sensorReading = null; 
303                          sensorReading = dbReading.getInstance(); 
304                          if (reading != null){ 
305                              // The array constructed from two arrays have the same length, Y, then X. 
306                              int middleOfArray = reading.length / 2; 
307                              // two arrays each one with half of the main array length 
308                              double[] x_reading = new double[middleOfArray], 
309                                      y_reading = new double[middleOfArray]; 
310                              // split the main array to two arrays, Y & X 
311                              for (int i = 0; i < middleOfArray; i++) { 
312                                  y_reading[i] = reading[i]; 
313                                  x_reading[i] = reading[middleOfArray + i]; 
314                              } 
315                              // Prepare the data to get ArrayRealVector with length 314 item, 
316                              // and set its value to the singleton sensor reading model. 
317                              sensorReading 
318                                      .setReading( 
319                                              new DataPreparation() 
320                                                      .getPreparedData( 
321                                                              x_reading, 
322                                                              y_reading 
323                                                      ) 
324                                      ); 
325                              // Notify the scan presenter with the new data 
326                              new ScanPresenter().gotSensorReading(mContext); 
327                              logMessage( 
328                                      "bluetooth..reading", 
329                                      "Sensor Reading, Length = " + 
330                                      sensorReading.getReading().getDimension() + 
331                                              ", Data: " + 
332                                              sensorReading.getReading().toString()); 
333   
334                          }else { 
335                              logMessage(TAG, "Reading equals null"); 
336                          } 
337                      } 
338                      break; 
339                  case "sensorWriting": 
340                      if (!intent.getBooleanExtra("isWriteSuccess", false)){ 
341                          logMessage("bluetooth .. write", "Writing process has failed"); 
342                      }else { 
343                          logMessage("bluetooth .. write", "Write process success"); 
344                      } 
345                      break; 
346                  default: 
347                      Log.v( TAG + "intent", "Got unknown broadcast intent"); 
348              } 
349          } 
350      };

376      private void askForCapturePermission() { 
377          MethodsFactory.logMessage(TAG, "Asking for capture permission"); 
378   
379          if (ContextCompat.checkSelfPermission(getActivity(), 
380                  Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) { 
381              if (getFromPref(getActivity(), ALLOW_KEY)) { 
382                  showSettingsAlert(); 
383              } else if (ContextCompat.checkSelfPermission(getActivity(), 
384                      Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) { 
385   
386                  // Should we show an explanation? 
387                  if (ActivityCompat.shouldShowRequestPermissionRationale(getActivity(), 
388                          Manifest.permission.CAMERA)) { 
389                      showAlert(); 
390                  } else { 
391                      // No explanation needed, we can request the permission. 
392                      ActivityCompat.requestPermissions(getActivity(), 
393                              new String[]{Manifest.permission.CAMERA}, 
394                              MY_PERMISSIONS_REQUEST_CAMERA); 
395                  } 
396              } 
397          } 
398      }

414      public static void  startInstalledAppDetailsActivity(@Nullable final Activity context, int code) { 
415          if (context == null) { 
416              return; 
417          } 
418          final Intent i = new Intent(MediaStore.ACTION_IMAGE_CAPTURE); 
419          i.putExtra("return-data", true); 
420          i.setAction(Settings.ACTION_APPLICATION_DETAILS_SETTINGS); 
421          i.addCategory(Intent.CATEGORY_DEFAULT); 
422          i.setData(Uri.parse("package:" + context.getPackageName())); 
423          i.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK); 
424          i.addFlags(Intent.FLAG_ACTIVITY_NO_HISTORY); 
425          i.addFlags(Intent.FLAG_ACTIVITY_EXCLUDE_FROM_RECENTS); 
426          context.startActivityForResult(i, code); 
427      }
