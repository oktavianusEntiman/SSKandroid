14    
15   import com.si_ware.neospectra.R;

17    
18   public class NavDrawerActivity extends AppCompatActivity 
19           implements NavigationView.OnNavigationItemSelectedListener { 
20    
21       public static final int GOTO_FRAGMENT_SCAN = 0; 
22       public static final int GOTO_FRAGMENT_MY_MODULES = 1; 
23       public static final int GOTO_FRAGMENT_SHOP = 2; 
24       public static final int GOTO_FRAGMENT_ABOUT_US = 3; 
25       public static final int GOTO_FRAGMENT_HISTORY = 4; 
26       NavigationView navigationView; 
27    
28       @Override 
29       protected void onCreate(Bundle savedInstanceState) { 
30           super.onCreate(savedInstanceState); 
31           setContentView(R.layout.activity_nav_drawer); 
32           Toolbar toolbar = findViewById(R.id.toolbar); 
33           setSupportActionBar(toolbar); 
34    
35           DrawerLayout drawer = findViewById(R.id.drawer_layout); 
36           ActionBarDrawerToggle toggle = new ActionBarDrawerToggle( 
37                   this, drawer, toolbar, R.string.navigation_drawer_open, R.string.navigation_drawer_close); 
38           drawer.addDrawerListener(toggle); 
39           toggle.syncState(); 
40    
41           navigationView = findViewById(R.id.nav_view); 
42           navigationView.setNavigationItemSelectedListener(this); 
43       } 
44    
45       @Override 
46       public void onBackPressed() { 
47           DrawerLayout drawer = findViewById(R.id.drawer_layout); 
48           if (drawer.isDrawerOpen(GravityCompat.START)) { 
49               drawer.closeDrawer(GravityCompat.START); 
50           } else { 
51               super.onBackPressed(); 
52           } 
53       } 
54    
55       @Override 
56       public boolean onOptionsItemSelected(@NonNull MenuItem item) { 
57           handleItemSelection(item); 
58           return super.onOptionsItemSelected(item); 
59       } 
60    
61       @Override 
62       public boolean onNavigationItemSelected(MenuItem item) { 
63           handleItemSelection(item); 
64           DrawerLayout drawer = findViewById(R.id.drawer_layout); 
65           drawer.closeDrawer(GravityCompat.START); 
66           return true; 
67       } 
68    
69       void handleItemSelection(@NonNull MenuItem item){ 
70           int id = item.getItemId(); 
71           if (id == R.id.nav_settings) { 
72               startActivity(new Intent(this, SettingsActivity.class)); 
73           } else if (id == R.id.nav_home) { 
74               startActivity(new Intent(this, MainActivity.class)); 
75           } 
76           else if (id == R.id.nav_new_modules) { 
77               Intent iBuyModules = new Intent(this, MainActivity.class); 
78               iBuyModules.putExtra("goto", GOTO_FRAGMENT_SHOP); 
79               startActivity(iBuyModules); 
80           } 
81           else if (id == R.id.nav_history) { 
82               Intent iHistory = new Intent(this, MainActivity.class); 
83               iHistory.putExtra("goto", GOTO_FRAGMENT_HISTORY); 
84               startActivity(iHistory); 
85           } 
86           else if(id == R.id.nav_about_us){ 
87               Log.v("btnPress", "pressed about from onNavigationItemSelected 2"); 
88               startActivity(new Intent(this, MainActivity.class).putExtra("goto", GOTO_FRAGMENT_ABOUT_US)); 
89           } 
90       } 
91   }
