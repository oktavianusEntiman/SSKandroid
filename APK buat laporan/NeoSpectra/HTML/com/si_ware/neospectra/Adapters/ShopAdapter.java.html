27   public class ShopAdapter extends RecyclerView.Adapter<ShopAdapter.MyViewHolder> { 
28    
29       private LayoutInflater inflater; 
30       private List<dbModuleInfo> modules; 
31       private Context mContext; 
32       @NonNull 
33       private String TAG = "ShopAdapter"; 
34    
35       public ShopAdapter(Context mContext, List<dbModuleInfo> list){ 
36           inflater = LayoutInflater.from(mContext); 
37           this.mContext = mContext; 
38           logMessage(TAG, "Constructor of shop adapter"); 
39           this.modules = list; 
40           logMessage(TAG, "List length: " + this.modules.size()); 
41       } 
42    
43       @NonNull 
44       @Override 
45       public MyViewHolder onCreateViewHolder(ViewGroup parent, int viewType) { 
46           View view = inflater.inflate(R.layout.row_buy_module, parent, false); 
47           return new MyViewHolder(view); 
48       } 
49    
50       @Override 
51       public void onBindViewHolder(@NonNull MyViewHolder currentViewHolder, int position) { 
52           logMessage(TAG, "On Bind View 0"); 
53           try { 
54               logMessage(TAG, "On Bind View 1"); 
55               dbModuleInfo moduleInfo = modules.get(currentViewHolder.getAdapterPosition()); 
56               logMessage(TAG, "Info: " + (moduleInfo == null)); 
57               logMessage(TAG, "Is Paid: " + moduleInfo.isPaidModule()); 
58               currentViewHolder.icon.setImageBitmap(moduleInfo.getIcon()); 
59               currentViewHolder.moduleName.setText(moduleInfo.getName()); 
60               currentViewHolder.price.setText(String.format("%s$", moduleInfo.getPrice())); 
61               if (moduleInfo.isPaidModule()) { 
62                   currentViewHolder.isPaid.setVisibility(View.VISIBLE); 
63                   currentViewHolder.buy.setVisibility(View.GONE); 
64               } 
65               logMessage(TAG, "On Bind View 2"); 
66    
67               currentViewHolder.rowItem.setOnClickListener(new View.OnClickListener() { 
68                   @Override 
69                   public void onClick(View v) { 
70                       Intent i = new Intent(mContext, ModuleDetailsActivity.class); 
71                       i.putExtra("from", "buy"); 
72   //                    i.putExtra("module", currentModule); 
73                       mContext.startActivity(i); 
74                   } 
75               }); 
76    
77           }catch (Exception e){ 
78               logMessage(TAG, e.getMessage()); 
79           } 
80       } 
81    
82       @Override 
83       public int getItemCount() { 
84           return modules.size(); 
85       } 
86    
87       class MyViewHolder extends RecyclerView.ViewHolder { 
88    
89           private ImageView icon, isPaid; 
90           private TextView moduleName, price; 
91           private ConstraintLayout buy, rowItem; 
92    
93           MyViewHolder(@NonNull View itemView) { 
94               super(itemView); 
95               icon = itemView.findViewById(R.id.iv_module_icon); 
96               moduleName = itemView.findViewById(R.id.tv_module_name); 
97               price = itemView.findViewById(R.id.tv_module_price); 
98               isPaid = itemView.findViewById(R.id.ic_module_paid); 
99               buy = itemView.findViewById(R.id.cl_buy_btn); 
100              rowItem = itemView.findViewById(R.id.cl_row_buy_module); 
101          } 
102      } 
103  }
