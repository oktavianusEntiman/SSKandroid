34   public class MyModulesAdapter extends RecyclerView.Adapter<MyModulesAdapter.MyViewHolder> { 
35    
36       private LayoutInflater inflater; 
37       private List<dbModuleInfo> modules; 
38       private int primaryModulePosition = -1; 
39       @NonNull 
40       private String TAG = "MyModulesAdapter"; 
41       private Context mContext; 
42    
43       public MyModulesAdapter(Context mContext, List<dbModuleInfo> list){ 
44           inflater = LayoutInflater.from(mContext); 
45           logMessage(TAG, "Constructor of my modules adapter"); 
46           this.modules = list; 
47           this.mContext = mContext; 
48           logMessage(TAG, "List length: " + this.modules.size()); 
49       } 
50    
51       @NonNull 
52       @Override 
53       public MyViewHolder onCreateViewHolder(ViewGroup parent, int viewType) { 
54           View view = inflater.inflate(R.layout.row_my_module, parent, false); 
55           return new MyViewHolder(view); 
56       } 
57    
58       @Override 
59       public void onBindViewHolder(@NonNull MyViewHolder currentViewHolder, int position) { 
60   //        logDebugging(TAG, "On Bind View 0"); 
61    
62           try { 
63   //            logDebugging(TAG, "On Bind View 1"); 
64               dbModuleInfo tempModule = modules.get(currentViewHolder.getAdapterPosition()); 
65    
66   //            logDebugging(TAG, "Info: " + (currentModule.getModuleInfo() == null)); 
67               currentViewHolder.icon.setImageBitmap(tempModule.getIcon()); 
68               currentViewHolder.moduleName.setText(tempModule.getName()); 
69               if (tempModule.getName().toLowerCase().equals(GlobalVariables.mySettings 
70                       .getPrimaryModuleName().toLowerCase())){ 
71                   currentViewHolder.isPrimary.setChecked(true); 
72                   primaryModulePosition = currentViewHolder.getAdapterPosition(); 
73                   // TODO: 6/1/18 Display message that there are missing values in the module info 
74                   // and ask the user to contact Si-Ware to provide it. 
75               } 
76   //            logDebugging(TAG, "On Bind View 2"); 
77           }catch (Exception e){ 
78   //            logDebugging(TAG, e.getMessage()); 
79           } 
80    
81           currentViewHolder.clMyModule.setOnClickListener(view -> { 
82               // Read this module and set it as default module 
83               currentModule = DataIO.retrieveDefaultModule(mContext, 
84                       modules.get(currentViewHolder.getAdapterPosition()).getName()); 
85    
86               mySettings.setPrimaryModuleName(currentModule.getModuleName()); 
87               // Inform the user that the primary module has changed 
88               Toast.makeText(mContext, 
89                       "The primary Module is: " + mySettings.getPrimaryModuleName(), 
90                       Toast.LENGTH_LONG).show(); 
91               gotoScan(); 
92           }); 
93       } 
94    
95       private void gotoScan(){ 
96           logMessage(TAG, "Goto Scan"); 
97           Intent iScan = new Intent(mContext, MainActivity.class); 
98           iScan.putExtra("goto", NavDrawerActivity.GOTO_FRAGMENT_SCAN); 
99           iScan.putExtra("moduleName", currentModule.getModuleName()); 
100          mContext.startActivity(iScan); 
101      } 
102   
103   
104      @Override 
105      public int getItemCount() { 
106          return modules.size(); 
107      }

113          private CheckBox isPrimary; 
114          private ConstraintLayout clMyModule; 
115

117              super(itemView); 
118              icon = itemView.findViewById(R.id.iv_module_icon); 
119              moduleName = itemView.findViewById(R.id.tv_module_name); 
120              isPrimary = itemView.findViewById(R.id.cb_is_primary_module); 
121              clMyModule = itemView.findViewById(R.id.cl_my_module); 
122          } 
123      }
