43   /** 
44    * Provides convenience methods for in-app billing. You can create one instance of this 
45    * class for your application and use it to process in-app billing operations. 
46    * It provides synchronous (blocking) and asynchronous (non-blocking) methods for 
47    * many common in-app billing operations, as well as automatic signature 
48    * verification. 
49    * 
50    * After instantiating, you must perform setup in order to start using the object. 
51    * To perform setup, call the {@link #startSetup} method and provide a listener; 
52    * that listener will be notified when setup is complete, after which (and not before) 
53    * you may call other methods. 
54    * 
55    * After setup is complete, you will typically want to request an inventory of owned 
56    * items and subscriptions. See {@link #queryInventory}, {@link #queryInventoryAsync} 
57    * and related methods. 
58    * 
59    * When you are done with this object, don't forget to call {@link #dispose} 
60    * to ensure proper cleanup. This object holds a binding to the in-app billing 
61    * service, which will leak unless you dispose of it correctly. If you created 
62    * the object on an Activity's onCreate method, then the recommended 
63    * place to dispose of it is the Activity's onDestroy method. It is invalid to 
64    * dispose the object while an asynchronous operation is in progress. You can 
65    * call {@link #disposeWhenFinished()} to ensure that any in-progress operation 
66    * completes before the object is disposed. 
67    * 
68    * A note about threading: When using this object from a background thread, you may 
69    * call the blocking versions of methods; when using from a UI thread, call 
70    * only the asynchronous versions and handle the results via callbacks. 
71    * Also, notice that you can only call one asynchronous operation at a time; 
72    * attempting to start a second asynchronous operation while the first one 
73    * has not yet completed will result in an exception being thrown. 
74    * 
75    */

216      /** 
217       * Starts the setup process. This will start up the setup process asynchronously. 
218       * You will be notified through the listener when the setup process is complete. 
219       * This method is safe to call from a UI thread. 
220       * 
221       * @param listener The listener to notify when the setup process is complete. 
222       */ 
223      public void startSetup(@Nullable final OnIabSetupFinishedListener listener) { 
224          // If already set up, can't do it again. 
225          checkNotDisposed(); 
226          if (mSetupDone) throw new IllegalStateException("IAB helper is already set up."); 
227   
228          // Connection to IAB service 
229          logDebug("Starting in-app billing setup."); 
230          mServiceConn = new ServiceConnection() { 
231              @Override 
232              public void onServiceDisconnected(ComponentName name) { 
233                  logDebug("Billing service disconnected."); 
234                  mService = null; 
235              } 
236   
237              @Override 
238              public void onServiceConnected(ComponentName name, IBinder service) { 
239                  if (mDisposed) return; 
240                  logDebug("Billing service connected."); 
241                  mService = IInAppBillingService.Stub.asInterface(service); 
242                  String packageName = mContext.getPackageName(); 
243                  try { 
244                      logDebug("Checking for in-app billing 3 support."); 
245   
246                      // check for in-app billing v3 support 
247                      int response = mService.isBillingSupported(3, packageName, ITEM_TYPE_INAPP); 
248                      if (response != BILLING_RESPONSE_RESULT_OK) { 
249                          if (listener != null) listener.onIabSetupFinished(new IabResult(response, 
250                                  "Error checking for billing v3 support.")); 
251   
252                          // if in-app purchases aren't supported, neither are subscriptions 
253                          mSubscriptionsSupported = false; 
254                          mSubscriptionUpdateSupported = false; 
255                          return; 
256                      } else { 
257                          logDebug("In-app billing version 3 supported for " + packageName); 
258                      } 
259   
260                      // Check for v5 subscriptions support. This is needed for 
261                      // getBuyIntentToReplaceSku which allows for subscription update 
262                      response = mService.isBillingSupported(5, packageName, ITEM_TYPE_SUBS); 
263                      if (response == BILLING_RESPONSE_RESULT_OK) { 
264                          logDebug("Subscription re-signup AVAILABLE."); 
265                          mSubscriptionUpdateSupported = true; 
266                      } else { 
267                          logDebug("Subscription re-signup not available."); 
268                          mSubscriptionUpdateSupported = false; 
269                      } 
270   
271                      if (mSubscriptionUpdateSupported) { 
272                          mSubscriptionsSupported = true; 
273                      } else { 
274                          // check for v3 subscriptions support 
275                          response = mService.isBillingSupported(3, packageName, ITEM_TYPE_SUBS); 
276                          if (response == BILLING_RESPONSE_RESULT_OK) { 
277                              logDebug("Subscriptions AVAILABLE."); 
278                              mSubscriptionsSupported = true; 
279                          } else { 
280                              logDebug("Subscriptions NOT AVAILABLE. Response: " + response); 
281                              mSubscriptionsSupported = false; 
282                              mSubscriptionUpdateSupported = false; 
283                          } 
284                      } 
285   
286                      mSetupDone = true; 
287                  } 
288                  catch (RemoteException e) { 
289                      if (listener != null) { 
290                          listener.onIabSetupFinished(new IabResult(IABHELPER_REMOTE_EXCEPTION, 
291                                  "RemoteException while setting up in-app billing.")); 
292                      } 
293                      e.printStackTrace(); 
294                      return; 
295                  } 
296   
297                  if (listener != null) { 
298                      listener.onIabSetupFinished(new IabResult(BILLING_RESPONSE_RESULT_OK, "Setup successful.")); 
299                  } 
300              } 
301          }; 
302   
303          Intent serviceIntent = new Intent("com.android.vending.billing.InAppBillingService.BIND"); 
304          serviceIntent.setPackage("com.android.vending"); 
305          List<ResolveInfo> intentServices = mContext.getPackageManager().queryIntentServices(serviceIntent, 0); 
306          if (intentServices != null && !intentServices.isEmpty()) { 
307              // service available to handle that Intent 
308              mContext.bindService(serviceIntent, mServiceConn, Context.BIND_AUTO_CREATE); 
309          } 
310          else { 
311              // no service available to handle that Intent 
312              if (listener != null) { 
313                  listener.onIabSetupFinished( 
314                          new IabResult(BILLING_RESPONSE_RESULT_BILLING_UNAVAILABLE, 
315                                  "Billing service unavailable on device.")); 
316              } 
317          } 
318      }

320      /** 
321       * Dispose of object, releasing resources. It's very important to call this 
322       * method when you are done with this object. It will release any resources 
323       * used by it such as service connections. Naturally, once the object is 
324       * disposed of, it can't be used again. 
325       */ 
326      public void dispose() throws IabAsyncInProgressException { 
327          synchronized (mAsyncInProgressLock) { 
328              if (mAsyncInProgress) { 
329                  throw new IabAsyncInProgressException("Can't dispose because an async operation " + 
330                          "(" + mAsyncOperation + ") is in progress."); 
331              } 
332          } 
333          logDebug("Disposing."); 
334          mSetupDone = false; 
335          if (mServiceConn != null) { 
336              logDebug("Unbinding from service."); 
337              if (mContext != null) mContext.unbindService(mServiceConn); 
338          } 
339          mDisposed = true; 
340          mContext = null; 
341          mServiceConn = null; 
342          mService = null; 
343          mPurchaseListener = null; 
344      } 
345   
346      /** 
347       * Disposes of object, releasing resources. If there is an in-progress async operation, this 
348       * method will queue the dispose to occur after the operation has finished. 
349       */ 
350      public void disposeWhenFinished() { 
351          synchronized (mAsyncInProgressLock) { 
352              if (mAsyncInProgress) { 
353                  logDebug("Will dispose after async operation finishes."); 
354                  mDisposeAfterAsync = true; 
355              } else { 
356                  try { 
357                      dispose(); 
358                  } catch (IabAsyncInProgressException e) { 
359                      // Should never be thrown, because we call dispose() only after checking that 
360                      // there's not already an async operation in progress. 
361                  } 
362              } 
363          } 
364      }

420      /** 
421       * Initiate the UI flow for an in-app purchase. Call this method to initiate an in-app purchase, 
422       * which will involve bringing up the Google Play screen. The calling activity will be paused 
423       * while the user interacts with Google Play, and the result will be delivered via the 
424       * activity's {@link android.app.Activity#onActivityResult} method, at which point you must call 
425       * this object's {@link #handleActivityResult} method to continue the purchase flow. This method 
426       * MUST be called from the UI thread of the Activity. 
427       * 
428       * @param act The calling activity. 
429       * @param sku The sku of the item to purchase. 
430       * @param itemType indicates if it's a product or a subscription (ITEM_TYPE_INAPP or 
431       *      ITEM_TYPE_SUBS) 
432       * @param oldSkus A list of SKUs which the new SKU is replacing or null if there are none 
433       * @param requestCode A request code (to differentiate from other responses -- as in 
434       *      {@link android.app.Activity#startActivityForResult}). 
435       * @param listener The listener to notify when the purchase process finishes 
436       * @param extraData Extra data (developer payload), which will be returned with the purchase 
437       *      data when the purchase completes. This extra data will be permanently bound to that 
438       *      purchase and will always be returned when the purchase is queried. 
439       */ 
440      public void launchPurchaseFlow(@NonNull Activity act, String sku, @NonNull String itemType, @Nullable List<String> oldSkus, 
441                                     int requestCode, @Nullable OnIabPurchaseFinishedListener listener, String extraData) 
442              throws IabAsyncInProgressException { 
443          checkNotDisposed(); 
444          checkSetupDone("launchPurchaseFlow"); 
445          flagStartAsync("launchPurchaseFlow"); 
446          IabResult result; 
447   
448          if (itemType.equals(ITEM_TYPE_SUBS) && !mSubscriptionsSupported) { 
449              IabResult r = new IabResult(IABHELPER_SUBSCRIPTIONS_NOT_AVAILABLE, 
450                      "Subscriptions are not available."); 
451              flagEndAsync(); 
452              if (listener != null) listener.onIabPurchaseFinished(r, null); 
453              return; 
454          } 
455   
456          try { 
457              logDebug("Constructing buy intent for " + sku + ", item type: " + itemType); 
458              Bundle buyIntentBundle; 
459              if (oldSkus == null || oldSkus.isEmpty()) { 
460                  // Purchasing a new item or subscription re-signup 
461                  buyIntentBundle = mService.getBuyIntent(3, mContext.getPackageName(), sku, itemType, 
462                          extraData); 
463              } else { 
464                  // Subscription upgrade/downgrade 
465                  if (!mSubscriptionUpdateSupported) { 
466                      IabResult r = new IabResult(IABHELPER_SUBSCRIPTION_UPDATE_NOT_AVAILABLE, 
467                              "Subscription updates are not available."); 
468                      flagEndAsync(); 
469                      if (listener != null) listener.onIabPurchaseFinished(r, null); 
470                      return; 
471                  } 
472                  buyIntentBundle = mService.getBuyIntentToReplaceSkus(5, mContext.getPackageName(), 
473                          oldSkus, sku, itemType, extraData); 
474              } 
475              int response = getResponseCodeFromBundle(buyIntentBundle); 
476              if (response != BILLING_RESPONSE_RESULT_OK) { 
477                  logError("Unable to buy item, Error response: " + getResponseDesc(response)); 
478                  flagEndAsync(); 
479                  result = new IabResult(response, "Unable to buy item"); 
480                  if (listener != null) listener.onIabPurchaseFinished(result, null); 
481                  return; 
482              } 
483   
484              PendingIntent pendingIntent = buyIntentBundle.getParcelable(RESPONSE_BUY_INTENT); 
485              logDebug("Launching buy intent for " + sku + ". Request code: " + requestCode); 
486              mRequestCode = requestCode; 
487              mPurchaseListener = listener; 
488              mPurchasingItemType = itemType; 
489              act.startIntentSenderForResult(pendingIntent.getIntentSender(), 
490                      requestCode, new Intent(), 
491                      Integer.valueOf(0), Integer.valueOf(0), 
492                      Integer.valueOf(0)); 
493          } 
494          catch (SendIntentException e) { 
495              logError("SendIntentException while launching purchase flow for sku " + sku); 
496              e.printStackTrace(); 
497              flagEndAsync(); 
498   
499              result = new IabResult(IABHELPER_SEND_INTENT_FAILED, "Failed to send intent."); 
500              if (listener != null) listener.onIabPurchaseFinished(result, null); 
501          } 
502          catch (RemoteException e) { 
503              logError("RemoteException while launching purchase flow for sku " + sku); 
504              e.printStackTrace(); 
505              flagEndAsync(); 
506   
507              result = new IabResult(IABHELPER_REMOTE_EXCEPTION, "Remote exception while starting purchase flow"); 
508              if (listener != null) listener.onIabPurchaseFinished(result, null); 
509          } 
510      }

512      /** 
513       * Handles an activity result that's part of the purchase flow in in-app billing. If you 
514       * are calling {@link #launchPurchaseFlow}, then you must call this method from your 
515       * Activity's {@link android.app.Activity@onActivityResult} method. This method 
516       * MUST be called from the UI thread of the Activity. 
517       * 
518       * @param requestCode The requestCode as you received it. 
519       * @param resultCode The resultCode as you received it. 
520       * @param data The data (Intent) as you received it. 
521       * @return Returns true if the result was related to a purchase flow and was handled; 
522       *     false if the result was not related to a purchase, in which case you should 
523       *     handle it normally. 
524       */ 
525      public boolean handleActivityResult(int requestCode, int resultCode, @Nullable Intent data) { 
526          IabResult result; 
527          if (requestCode != mRequestCode) return false; 
528   
529          checkNotDisposed(); 
530          checkSetupDone("handleActivityResult"); 
531   
532          // end of async purchase operation that started on launchPurchaseFlow 
533          flagEndAsync(); 
534   
535          if (data == null) { 
536              logError("Null data in IAB activity result."); 
537              result = new IabResult(IABHELPER_BAD_RESPONSE, "Null data in IAB result"); 
538              if (mPurchaseListener != null) mPurchaseListener.onIabPurchaseFinished(result, null); 
539              return true; 
540          } 
541   
542          int responseCode = getResponseCodeFromIntent(data); 
543          String purchaseData = data.getStringExtra(RESPONSE_INAPP_PURCHASE_DATA); 
544          String dataSignature = data.getStringExtra(RESPONSE_INAPP_SIGNATURE); 
545   
546          if (resultCode == Activity.RESULT_OK && responseCode == BILLING_RESPONSE_RESULT_OK) { 
547              logDebug("Successful resultcode from purchase activity."); 
548              logDebug("Purchase data: " + purchaseData); 
549              logDebug("Data signature: " + dataSignature); 
550              logDebug("Extras: " + data.getExtras()); 
551              logDebug("Expected item type: " + mPurchasingItemType); 
552   
553              if (purchaseData == null || dataSignature == null) { 
554                  logError("BUG: either purchaseData or dataSignature is null."); 
555                  logDebug("Extras: " + data.getExtras().toString()); 
556                  result = new IabResult(IABHELPER_UNKNOWN_ERROR, "IAB returned null purchaseData or dataSignature"); 
557                  if (mPurchaseListener != null) mPurchaseListener.onIabPurchaseFinished(result, null); 
558                  return true; 
559              } 
560   
561              Purchase purchase = null; 
562              try { 
563                  purchase = new Purchase(mPurchasingItemType, purchaseData, dataSignature); 
564                  String sku = purchase.getSku(); 
565   
566                  // Verify signature 
567                  if (!Security.verifyPurchase(mSignatureBase64, purchaseData, dataSignature)) { 
568                      logError("Purchase signature verification FAILED for sku " + sku); 
569                      result = new IabResult(IABHELPER_VERIFICATION_FAILED, "Signature verification failed for sku " + sku); 
570                      if (mPurchaseListener != null) mPurchaseListener.onIabPurchaseFinished(result, purchase); 
571                      return true; 
572                  } 
573                  logDebug("Purchase signature successfully verified."); 
574              } 
575              catch (JSONException e) { 
576                  logError("Failed to parse purchase data."); 
577                  e.printStackTrace(); 
578                  result = new IabResult(IABHELPER_BAD_RESPONSE, "Failed to parse purchase data."); 
579                  if (mPurchaseListener != null) mPurchaseListener.onIabPurchaseFinished(result, null); 
580                  return true; 
581              } 
582   
583              if (mPurchaseListener != null) { 
584                  mPurchaseListener.onIabPurchaseFinished(new IabResult(BILLING_RESPONSE_RESULT_OK, "Success"), purchase); 
585              } 
586          } 
587          else if (resultCode == Activity.RESULT_OK) { 
588              // result code was OK, but in-app billing response was not OK. 
589              logDebug("Result code was OK but in-app billing response was not OK: " + getResponseDesc(responseCode)); 
590              if (mPurchaseListener != null) { 
591                  result = new IabResult(responseCode, "Problem purchashing item."); 
592                  mPurchaseListener.onIabPurchaseFinished(result, null); 
593              } 
594          } 
595          else if (resultCode == Activity.RESULT_CANCELED) { 
596              logDebug("Purchase canceled - Response: " + getResponseDesc(responseCode)); 
597              result = new IabResult(IABHELPER_USER_CANCELLED, "User canceled."); 
598              if (mPurchaseListener != null) mPurchaseListener.onIabPurchaseFinished(result, null); 
599          } 
600          else { 
601              logError("Purchase failed. Result code: " + Integer.toString(resultCode) 
602                      + ". Response: " + getResponseDesc(responseCode)); 
603              result = new IabResult(IABHELPER_UNKNOWN_PURCHASE_RESPONSE, "Unknown purchase response."); 
604              if (mPurchaseListener != null) mPurchaseListener.onIabPurchaseFinished(result, null); 
605          } 
606          return true; 
607      }

614      /** 
615       * Queries the inventory. This will query all owned items from the server, as well as 
616       * information on additional skus, if specified. This method may block or take long to execute. 
617       * Do not call from a UI thread. For that, use the non-blocking version {@link #queryInventoryAsync}. 
618       * 
619       * @param querySkuDetails if true, SKU details (price, description, etc) will be queried as well 
620       *     as purchase information. 
621       * @param moreItemSkus additional PRODUCT skus to query information on, regardless of ownership. 
622       *     Ignored if null or if querySkuDetails is false. 
623       * @param moreSubsSkus additional SUBSCRIPTIONS skus to query information on, regardless of ownership. 
624       *     Ignored if null or if querySkuDetails is false. 
625       * @throws IabException if a problem occurs while refreshing the inventory. 
626       */ 
627      @NonNull 
628      public Inventory queryInventory(boolean querySkuDetails, List<String> moreItemSkus, 
629                                      List<String> moreSubsSkus) throws IabException { 
630          checkNotDisposed(); 
631          checkSetupDone("queryInventory"); 
632          try { 
633              Inventory inv = new Inventory(); 
634              int r = queryPurchases(inv, ITEM_TYPE_INAPP); 
635              if (r != BILLING_RESPONSE_RESULT_OK) { 
636                  throw new IabException(r, "Error refreshing inventory (querying owned items)."); 
637              } 
638   
639              if (querySkuDetails) { 
640                  r = querySkuDetails(ITEM_TYPE_INAPP, inv, moreItemSkus); 
641                  if (r != BILLING_RESPONSE_RESULT_OK) { 
642                      throw new IabException(r, "Error refreshing inventory (querying prices of items)."); 
643                  } 
644              } 
645   
646              // if subscriptions are supported, then also query for subscriptions 
647              if (mSubscriptionsSupported) { 
648                  r = queryPurchases(inv, ITEM_TYPE_SUBS); 
649                  if (r != BILLING_RESPONSE_RESULT_OK) { 
650                      throw new IabException(r, "Error refreshing inventory (querying owned subscriptions)."); 
651                  } 
652   
653                  if (querySkuDetails) { 
654                      r = querySkuDetails(ITEM_TYPE_SUBS, inv, moreSubsSkus); 
655                      if (r != BILLING_RESPONSE_RESULT_OK) { 
656                          throw new IabException(r, "Error refreshing inventory (querying prices of subscriptions)."); 
657                      } 
658                  } 
659              } 
660   
661              return inv; 
662          } 
663          catch (RemoteException e) { 
664              throw new IabException(IABHELPER_REMOTE_EXCEPTION, "Remote exception while refreshing inventory.", e); 
665          } 
666          catch (JSONException e) { 
667              throw new IabException(IABHELPER_BAD_RESPONSE, "Error parsing JSON response while refreshing inventory.", e); 
668          } 
669      }

734      /** 
735       * Consumes a given in-app product. Consuming can only be done on an item 
736       * that's owned, and as a result of consumption, the user will no longer own it. 
737       * This method may block or take long to return. Do not call from the UI thread. 
738       * For that, see {@link #consumeAsync}. 
739       * 
740       * @param itemInfo The PurchaseInfo that represents the item to consume. 
741       * @throws IabException if there is a problem during consumption. 
742       */ 
743      void consume(@NonNull Purchase itemInfo) throws IabException { 
744          checkNotDisposed(); 
745          checkSetupDone("consume"); 
746   
747          if (!itemInfo.mItemType.equals(ITEM_TYPE_INAPP)) { 
748              throw new IabException(IABHELPER_INVALID_CONSUMPTION, 
749                      "Items of type '" + itemInfo.mItemType + "' can't be consumed."); 
750          } 
751   
752          try { 
753              String token = itemInfo.getToken(); 
754              String sku = itemInfo.getSku(); 
755              if (token == null || token.equals("")) { 
756                  logError("Can't consume "+ sku + ". No token."); 
757                  throw new IabException(IABHELPER_MISSING_TOKEN, "PurchaseInfo is missing token for sku: " 
758                          + sku + " " + itemInfo); 
759              } 
760   
761              logDebug("Consuming sku: " + sku + ", token: " + token); 
762              int response = mService.consumePurchase(3, mContext.getPackageName(), token); 
763              if (response == BILLING_RESPONSE_RESULT_OK) { 
764                  logDebug("Successfully consumed sku: " + sku); 
765              } 
766              else { 
767                  logDebug("Error consuming consuming sku " + sku + ". " + getResponseDesc(response)); 
768                  throw new IabException(response, "Error consuming sku " + sku); 
769              } 
770          } 
771          catch (RemoteException e) { 
772              throw new IabException(IABHELPER_REMOTE_EXCEPTION, "Remote exception while consuming. PurchaseInfo: " + itemInfo, e); 
773          } 
774      }

832      /** 
833       * Returns a human-readable description for the given response code. 
834       * 
835       * @param code The response code 
836       * @return A human-readable string explaining the result code. 
837       *     It also includes the result code numerically. 
838       */ 
839      public static String getResponseDesc(int code) { 
840          String[] iab_msgs = ("0:OK/1:User Canceled/2:Unknown/" + 
841                  "3:Billing Unavailable/4:Item unavailable/" + 
842                  "5:Developer Error/6:Error/7:Item Already Owned/" + 
843                  "8:Item not owned").split("/"); 
844          String[] iabhelper_msgs = ("0:OK/-1001:Remote exception during initialization/" + 
845                  "-1002:Bad response received/" + 
846                  "-1003:Purchase signature verification failed/" + 
847                  "-1004:Send intent failed/" + 
848                  "-1005:User cancelled/" + 
849                  "-1006:Unknown purchase response/" + 
850                  "-1007:Missing token/" + 
851                  "-1008:Unknown error/" + 
852                  "-1009:Subscriptions not available/" + 
853                  "-1010:Invalid consumption attempt").split("/"); 
854   
855          if (code <= IABHELPER_ERROR_BASE) { 
856              int index = IABHELPER_ERROR_BASE - code; 
857              if (index >= 0 && index < iabhelper_msgs.length) return iabhelper_msgs[index]; 
858              else return String.valueOf(code) + ":Unknown IAB Helper Error"; 
859          } 
860          else if (code < 0 || code >= iab_msgs.length) 
861              return String.valueOf(code) + ":Unknown"; 
862          else 
863              return iab_msgs[code]; 
864      }

875      // Workaround to bug where sometimes response codes come as Long instead of Integer 
876      int getResponseCodeFromBundle(@NonNull Bundle b) { 
877          Object o = b.get(RESPONSE_CODE); 
878          if (o == null) { 
879              logDebug("Bundle with null response code, assuming OK (known issue)"); 
880              return BILLING_RESPONSE_RESULT_OK; 
881          } 
882          else if (o instanceof Integer) return ((Integer)o).intValue(); 
883          else if (o instanceof Long) return (int)((Long)o).longValue(); 
884          else { 
885              logError("Unexpected type for bundle response code."); 
886              logError(o.getClass().getName()); 
887              throw new RuntimeException("Unexpected type for bundle response code: " + o.getClass().getName()); 
888          } 
889      }

907      void flagStartAsync(String operation) throws IabAsyncInProgressException { 
908          synchronized (mAsyncInProgressLock) { 
909              if (mAsyncInProgress) { 
910                  throw new IabAsyncInProgressException("Can't start async operation (" + 
911                          operation + ") because another async operation (" + mAsyncOperation + 
912                          ") is in progress."); 
913              } 
914              mAsyncOperation = operation; 
915              mAsyncInProgress = true; 
916              logDebug("Starting async operation: " + operation); 
917          } 
918      }

920      void flagEndAsync() { 
921          synchronized (mAsyncInProgressLock) { 
922              logDebug("Ending async operation: " + mAsyncOperation); 
923              mAsyncOperation = ""; 
924              mAsyncInProgress = false; 
925              if (mDisposeAfterAsync) { 
926                  try { 
927                      dispose(); 
928                  } catch (IabAsyncInProgressException e) { 
929                      // Should not be thrown, because we reset mAsyncInProgress immediately before 
930                      // calling dispose(). 
931                  } 
932              } 
933          } 
934      }

946      int queryPurchases(@NonNull Inventory inv, String itemType) throws JSONException, RemoteException { 
947          // Query purchases 
948          logDebug("Querying owned items, item type: " + itemType); 
949          logDebug("Package name: " + mContext.getPackageName()); 
950          boolean verificationFailed = false; 
951          String continueToken = null; 
952   
953          do { 
954              logDebug("Calling getPurchases with continuation token: " + continueToken); 
955              Bundle ownedItems = mService.getPurchases(3, mContext.getPackageName(), 
956                      itemType, continueToken); 
957   
958              int response = getResponseCodeFromBundle(ownedItems); 
959              logDebug("Owned items response: " + String.valueOf(response)); 
960              if (response != BILLING_RESPONSE_RESULT_OK) { 
961                  logDebug("getPurchases() failed: " + getResponseDesc(response)); 
962                  return response; 
963              } 
964              if (!ownedItems.containsKey(RESPONSE_INAPP_ITEM_LIST) 
965                      || !ownedItems.containsKey(RESPONSE_INAPP_PURCHASE_DATA_LIST) 
966                      || !ownedItems.containsKey(RESPONSE_INAPP_SIGNATURE_LIST)) { 
967                  logError("Bundle returned from getPurchases() doesn't contain required fields."); 
968                  return IABHELPER_BAD_RESPONSE; 
969              } 
970   
971              ArrayList<String> ownedSkus = ownedItems.getStringArrayList( 
972                      RESPONSE_INAPP_ITEM_LIST); 
973              ArrayList<String> purchaseDataList = ownedItems.getStringArrayList( 
974                      RESPONSE_INAPP_PURCHASE_DATA_LIST); 
975              ArrayList<String> signatureList = ownedItems.getStringArrayList( 
976                      RESPONSE_INAPP_SIGNATURE_LIST); 
977   
978              for (int i = 0; i < purchaseDataList.size(); ++i) { 
979                  String purchaseData = purchaseDataList.get(i); 
980                  String signature = signatureList.get(i); 
981                  String sku = ownedSkus.get(i); 
982                  if (Security.verifyPurchase(mSignatureBase64, purchaseData, signature)) { 
983                      logDebug("Sku is owned: " + sku); 
984                      Purchase purchase = new Purchase(itemType, purchaseData, signature); 
985   
986                      if (TextUtils.isEmpty(purchase.getToken())) { 
987                          logWarn("BUG: empty/null token!"); 
988                          logDebug("Purchase data: " + purchaseData); 
989                      } 
990   
991                      // Record ownership and token 
992                      inv.addPurchase(purchase); 
993                  } 
994                  else { 
995                      logWarn("Purchase signature verification **FAILED**. Not adding item."); 
996                      logDebug("   Purchase data: " + purchaseData); 
997                      logDebug("   Signature: " + signature); 
998                      verificationFailed = true; 
999                  } 
1000             } 
1001  
1002             continueToken = ownedItems.getString(INAPP_CONTINUATION_TOKEN); 
1003             logDebug("Continuation token: " + continueToken); 
1004         } while (!TextUtils.isEmpty(continueToken)); 
1005  
1006         return verificationFailed ? IABHELPER_VERIFICATION_FAILED : BILLING_RESPONSE_RESULT_OK; 
1007     }

1009     int querySkuDetails(String itemType, @NonNull Inventory inv, @Nullable List<String> moreSkus) 
1010             throws RemoteException, JSONException { 
1011         logDebug("Querying SKU details."); 
1012         ArrayList<String> skuList = new ArrayList<String>(); 
1013         skuList.addAll(inv.getAllOwnedSkus(itemType)); 
1014         if (moreSkus != null) { 
1015             for (String sku : moreSkus) { 
1016                 if (!skuList.contains(sku)) { 
1017                     skuList.add(sku); 
1018                 } 
1019             } 
1020         } 
1021  
1022         if (skuList.size() == 0) { 
1023             logDebug("queryPrices: nothing to do because there are no SKUs."); 
1024             return BILLING_RESPONSE_RESULT_OK; 
1025         } 
1026  
1027         // Split the sku list in blocks of no more than 20 elements. 
1028         ArrayList<ArrayList<String>> packs = new ArrayList<ArrayList<String>>(); 
1029         ArrayList<String> tempList; 
1030         int n = skuList.size() / 20; 
1031         int mod = skuList.size() % 20; 
1032         for (int i = 0; i < n; i++) { 
1033             tempList = new ArrayList<String>(); 
1034             for (String s : skuList.subList(i * 20, i * 20 + 20)) { 
1035                 tempList.add(s); 
1036             } 
1037             packs.add(tempList); 
1038         } 
1039         if (mod != 0) { 
1040             tempList = new ArrayList<String>(); 
1041             for (String s : skuList.subList(n * 20, n * 20 + mod)) { 
1042                 tempList.add(s); 
1043             } 
1044             packs.add(tempList); 
1045         } 
1046  
1047         for (ArrayList<String> skuPartList : packs) { 
1048             Bundle querySkus = new Bundle(); 
1049             querySkus.putStringArrayList(GET_SKU_DETAILS_ITEM_LIST, skuPartList); 
1050             Bundle skuDetails = mService.getSkuDetails(3, mContext.getPackageName(), 
1051                     itemType, querySkus); 
1052  
1053             if (!skuDetails.containsKey(RESPONSE_GET_SKU_DETAILS_LIST)) { 
1054                 int response = getResponseCodeFromBundle(skuDetails); 
1055                 if (response != BILLING_RESPONSE_RESULT_OK) { 
1056                     logDebug("getSkuDetails() failed: " + getResponseDesc(response)); 
1057                     return response; 
1058                 } else { 
1059                     logError("getSkuDetails() returned a bundle with neither an error nor a detail list."); 
1060                     return IABHELPER_BAD_RESPONSE; 
1061                 } 
1062             } 
1063  
1064             ArrayList<String> responseList = skuDetails.getStringArrayList( 
1065                     RESPONSE_GET_SKU_DETAILS_LIST); 
1066  
1067             for (String thisResponse : responseList) { 
1068                 SkuDetails d = new SkuDetails(itemType, thisResponse); 
1069                 logDebug("Got sku details: " + d); 
1070                 inv.addSkuDetails(d); 
1071             } 
1072         } 
1073  
1074         return BILLING_RESPONSE_RESULT_OK; 
1075     }

1077     void consumeAsyncInternal(@NonNull final List<Purchase> purchases, 
1078                               @Nullable final OnConsumeFinishedListener singleListener, 
1079                               @Nullable final OnConsumeMultiFinishedListener multiListener) 
1080             throws IabAsyncInProgressException { 
1081         final Handler handler = new Handler(); 
1082         flagStartAsync("consume"); 
1083         (new Thread(new Runnable() { 
1084             public void run() { 
1085                 final List<IabResult> results = new ArrayList<IabResult>(); 
1086                 for (Purchase purchase : purchases) { 
1087                     try { 
1088                         consume(purchase); 
1089                         results.add(new IabResult(BILLING_RESPONSE_RESULT_OK, "Successful consume of sku " + purchase.getSku())); 
1090                     } 
1091                     catch (IabException ex) { 
1092                         results.add(ex.getResult()); 
1093                     } 
1094                 } 
1095  
1096                 flagEndAsync(); 
1097                 if (!mDisposed && singleListener != null) { 
1098                     handler.post(new Runnable() { 
1099                         public void run() { 
1100                             singleListener.onConsumeFinished(purchases.get(0), results.get(0)); 
1101                         } 
1102                     }); 
1103                 } 
1104                 if (!mDisposed && multiListener != null) { 
1105                     handler.post(new Runnable() { 
1106                         public void run() { 
1107                             multiListener.onConsumeMultiFinished(purchases, results); 
1108                         } 
1109                     }); 
1110                 } 
1111             } 
1112         })).start(); 
1113     }
