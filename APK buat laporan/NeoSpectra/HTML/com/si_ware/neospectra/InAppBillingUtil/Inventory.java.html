16   package com.si_ware.neospectra.InAppBillingUtil; 
17    
18   import android.support.annotation.NonNull; 
19    
20   import java.util.ArrayList; 
21   import java.util.HashMap; 
22   import java.util.List; 
23   import java.util.Map; 
24    
25   /** 
26    * Represents a block of information about in-app items. 
27    * An Inventory is returned by such methods as {@link IabHelper#queryInventory}. 
28    */ 
29   public class Inventory { 
30       @NonNull 
31       Map<String,SkuDetails> mSkuMap = new HashMap<String,SkuDetails>(); 
32       @NonNull 
33       Map<String,Purchase> mPurchaseMap = new HashMap<String,Purchase>(); 
34    
35       Inventory() { } 
36    
37       /** Returns the listing details for an in-app product. */ 
38       public SkuDetails getSkuDetails(String sku) { 
39           return mSkuMap.get(sku); 
40       } 
41    
42       /** Returns purchase information for a given product, or null if there is no purchase. */ 
43       public Purchase getPurchase(String sku) { 
44           return mPurchaseMap.get(sku); 
45       } 
46    
47       /** Returns whether or not there exists a purchase of the given product. */ 
48       public boolean hasPurchase(String sku) { 
49           return mPurchaseMap.containsKey(sku); 
50       } 
51    
52       /** Return whether or not details about the given product are available. */ 
53       public boolean hasDetails(String sku) { 
54           return mSkuMap.containsKey(sku); 
55       } 
56    
57       /** 
58        * Erase a purchase (locally) from the inventory, given its product ID. This just 
59        * modifies the Inventory object locally and has no effect on the server! This is 
60        * useful when you have an existing Inventory object which you know to be up to date, 
61        * and you have just consumed an item successfully, which means that erasing its 
62        * purchase data from the Inventory you already have is quicker than querying for 
63        * a new Inventory. 
64        */ 
65       public void erasePurchase(String sku) { 
66           if (mPurchaseMap.containsKey(sku)) mPurchaseMap.remove(sku); 
67       } 
68    
69       /** Returns a list of all owned product IDs. */ 
70       @NonNull 
71       List<String> getAllOwnedSkus() { 
72           return new ArrayList<String>(mPurchaseMap.keySet()); 
73       } 
74    
75       /** Returns a list of all owned product IDs of a given type */ 
76       @NonNull 
77       List<String> getAllOwnedSkus(String itemType) { 
78           List<String> result = new ArrayList<String>(); 
79           for (Purchase p : mPurchaseMap.values()) { 
80               if (p.getItemType().equals(itemType)) result.add(p.getSku()); 
81           } 
82           return result; 
83       } 
84    
85       /** Returns a list of all purchases. */ 
86       @NonNull 
87       List<Purchase> getAllPurchases() { 
88           return new ArrayList<Purchase>(mPurchaseMap.values()); 
89       } 
90    
91       void addSkuDetails(@NonNull SkuDetails d) { 
92           mSkuMap.put(d.getSku(), d); 
93       } 
94    
95       void addPurchase(@NonNull Purchase p) { 
96           mPurchaseMap.put(p.getSku(), p); 
97       } 
98   }
