16   package com.si_ware.neospectra.InAppBillingUtil; 
17    
18   import android.support.annotation.NonNull; 
19   import android.support.annotation.Nullable; 
20    
21   /** 
22    * Represents the result of an in-app billing operation. 
23    * A result is composed of a response code (an integer) and possibly a 
24    * message (String). You can get those by calling 
25    * {@link #getResponse} and {@link #getMessage()}, respectively. You 
26    * can also inquire whether a result is a success or a failure by 
27    * calling {@link #isSuccess()} and {@link #isFailure()}. 
28    */ 
29   public class IabResult { 
30       int mResponse; 
31       String mMessage; 
32    
33       public IabResult(int response, @Nullable String message) { 
34           mResponse = response; 
35           if (message == null || message.trim().length() == 0) { 
36               mMessage = IabHelper.getResponseDesc(response); 
37           } 
38           else { 
39               mMessage = message + " (response: " + IabHelper.getResponseDesc(response) + ")"; 
40           } 
41       } 
42       public int getResponse() { return mResponse; } 
43       public String getMessage() { return mMessage; } 
44       public boolean isSuccess() { return mResponse == IabHelper.BILLING_RESPONSE_RESULT_OK; } 
45       public boolean isFailure() { return !isSuccess(); } 
46       @NonNull 
47       public String toString() { return "IabResult: " + getMessage(); } 
48   }
