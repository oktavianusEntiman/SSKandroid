21       public void showResults(@NonNull Context mContext, @NonNull dbResult result, @NonNull List<ProgressBar> resultBars) { 
22           if (result.getResults().size() == resultBars.size()){ 
23               for (int i = 0; i < resultBars.size(); i++){ 
24                   animateResult( 
25   //                        provide the current progressbar 
26                           resultBars.get(i), 
27   //                        set duration 
28                           4000 + i * 300, 
29   //                        Variable name 
30                           result.getResults().get(i).getVarName(), 
31   //                        Get the result and cast it from double to float 
32                           (result.getResults().get(i).getVarResult().floatValue())); 
33               } 
34               askToSaveResults(mContext, result); 
35           }else { 
36               Log.e("err_NotMatching", mContext.getString(R.string.err_sizes_not_matching)); 
37           } 
38       } 
39    
40       public void askToSaveResults(@NonNull final Context mContext, final dbResult result){ 
41           String title = mContext.getString(R.string.title_save_results); 
42           String body = mContext.getString(R.string.ask_to_save_results); 
43           android.support.v7.app.AlertDialog.Builder myAlert = new android.support.v7.app.AlertDialog 
44                   .Builder(mContext); 
45           myAlert.setTitle(title); 
46           myAlert.setMessage(body); 
47           myAlert.setPositiveButton("Yes", (dialogInterface, i) -> { 
48               ScanPresenter presenter = new ScanPresenter(); 
49               presenter.saveResult(mContext, result); 
50               dialogInterface.dismiss(); 
51           }); 
52           myAlert.setNegativeButton("No", (dialog, which) -> dialog.dismiss()); 
53           myAlert.show(); 
54       } 
55    
56       public void animateResult(@NonNull ProgressBar progressBar, int duration, String varName, float result) { 
57           ProgressBarAnimation anim; 
58           anim = new ProgressBarAnimation(progressBar, 0, result); 
59           anim.setDuration(duration); 
60           progressBar.startAnimation(anim); 
61       }
