79       @NonNull 
80       @Override 
81       public double[][] emsc(@NonNull double[][] xData, int order) { 
82    
83           double[][] correct = new double[xData.length][xData[0].length]; 
84           double[] x = new double[xData[0].length]; 
85    
86           for (int i = 0; i < xData[0].length; i++) { 
87               double sum = 0.0; 
88               for (int j = 0; j < xData.length; j++) { 
89                   sum += xData[j][i]; 
90               } 
91               x[i] = sum / xData.length; 
92           } 
93    
94           for (int i = 0; i < xData.length; i++) { 
95    
96               double[][] y = new double[xData[0].length][1]; 
97               for (int j = 0; j < y.length; j++) { 
98                   y[j][0] = xData[i][j]; 
99               } 
100              double[][] X; 
101   
102              if (order > 0) { 
103                  X = new double[xData[0].length][order + 2]; 
104                  double[][] s = new double[xData[0].length][order + 1]; 
105                  for (int j = 0; j < s.length; j++) { 
106                      s[j][0] = 1.0; 
107                  } 
108   
109                  for (int j = 0; j < order; j++) { 
110                      double[] t = new double[xData[0].length]; 
111   
112                      double r = 0; 
113                      int k = 0; 
114   
115                      while (r < (1 + 1.0 / (y.length - 1))) { 
116                          t[k] = r; 
117                          r = r + 1.0 / ((y.length) - 1); 
118                          k++; 
119                      } 
120   
121                      for (int k2 = 0; k2 < t.length; k2++) { 
122                          t[k2] = Math.pow(t[k2], j); 
123                      } 
124   
125                      for (int k2 = 0; k2 < t.length; k2++) { 
126                          s[k2][j + 1] = t[k2]; 
127                      } 
128   
129                  } 
130   
131                  // X=np.concatenate((x, s),1) 
132                  for (int j = 0; j < X[0].length; j++) { 
133                      for (int j2 = 0; j2 < X.length; j2++) { 
134                          if (j == 0) { 
135                              X[j2][j] = x[j2]; 
136                          } else { 
137                              X[j2][j] = s[j2][j - 1]; 
138                          } 
139                      } 
140                  } 
141   
142              } else { 
143                  X = new double[xData[0].length][1]; 
144              } 
145   
146              // //calc fit b=fit coefficients 
147              Matrix XX = new Matrix(X); 
148              Matrix yy = new Matrix(y); 
149   
150              // b = 
151              // np.dot(np.dot(np.linalg.pinv(np.dot(scipy.transpose(X),X)),np.transpose(X)),y) 
152   
153              Matrix s1 = XX.transpose().times(XX); 
154   
155              Matrix s2= PseudoInverse.test(s1); 
156   
157              Matrix s3 = s2.times(XX.transpose()); 
158              Matrix bb = s3.times(yy); 
159              // f = np.dot(X,b) 
160              Matrix ff = XX.times(bb); 
161              double[][] f = ff.getArray(); 
162              double[][] b = bb.getArray(); 
163              // r = y - f 
164              double[] r = new double[y.length]; 
165              for (int j = 0; j < X.length; j++) { 
166                  r[j] = y[j][0] - f[j][0]; 
167              } 
168   
169              for (int j = 0; j < r.length; j++) { 
170                  correct[i][j] = (r[j] / b[0][0]) + x[j]; 
171              } 
172          } 
173          return (correct); 
174      }
