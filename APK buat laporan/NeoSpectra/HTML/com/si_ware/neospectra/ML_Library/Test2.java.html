54       public static double svrPrediction( 
55               ArrayRealVector v, 
56               String fileName, 
57               @NonNull ArrayRealVector x_mean, 
58               ArrayRealVector x_std, 
59               double y_std, 
60               double y_mean 
61       ) throws IOException { 
62    
63           svm_model current_model = svm.svm_load_model(fileName); 
64   //        svm_model current_model = svm.svm_load_model(loadModel(fileName)); 
65           ArrayRealVector normalized_input = v.subtract(x_mean); 
66           normalized_input = normalized_input.ebeDivide(x_std); 
67    
68           double[] fVector = normalized_input.toArray(); 
69           svm_node[] nodes = new svm_node[fVector.length]; 
70           for (int i = 0; i < fVector.length; i++) 
71           { 
72               svm_node node = new svm_node(); 
73               node.index = i; 
74               node.value = fVector[i]; 
75               nodes[i] = node; 
76           } 
77    
78           double result =svm.svm_predict(current_model,nodes); 
79           result*=y_std; 
80           result+=y_mean; 
81           return result; 
82       }

85       public static double gprPrediction(ArrayRealVector v, 
86                                          RealMatrix trainingData, 
87                                          double length_scale, 
88                                          ArrayRealVector alpha, 
89                                          double y_train_mean) 
90       {   //procedures 
91           //euclidean distance with training data (1x314)(200X314) note that both should be divided by the length scale== > (200x314) 
92           //k = exp(-0.5*the distance calculated) 
93           //return k 
94           //dot product with self.alpha_ matrix (1x200)(200,1) == (1) 
95           //add to it mean of y train 
96           //return result 
97    
98           v.mapDivideToSelf(length_scale); 
99           trainingData = trainingData.scalarMultiply(1.0/length_scale); 
100          double[] sqEucDist = new double[trainingData.getRowDimension()]; 
101          double dist; 
102          for(int i=0;i<sqEucDist.length;i++) 
103          { 
104              dist = v.getDistance(trainingData.getRowVector(i)); 
105              dist=dist*dist; 
106              //apply exponentiation 
107              sqEucDist[i] = Math.exp(-0.5*dist); 
108          } 
109          ArrayRealVector k=new ArrayRealVector(sqEucDist); 
110          double ymean = k.dotProduct(alpha); 
111          ymean+=y_train_mean; 
112          return ymean; 
113      }

117                                             RealMatrix W, 
118                                             @NonNull RealMatrix P) 
119      {

121          ArrayRealVector W_column,P_column; 
122          double t; 
123          for(int i =0 ;i <k ;i++) 
124          { 
125              W_column =new ArrayRealVector( W.getColumn(k)); 
126              P_column = new ArrayRealVector(P.getColumn(k)); 
127              t = vector.dotProduct(W_column);

383      { 
384          double[] original = v.toArray(); 
385          double[] output = new double[original.length-1]; 
386          output[0]=0; 
387          double[] intermediate; 
388          double j=0; 
389          for(int i=1;i< original.length-1;i++) 
390          { 
391              intermediate = Arrays.copyOfRange(original, 0, i+1); 
392              j=trapzRule(intermediate); 
393              output[i]=j; 
394   
395          } 
396   
397   
398          return new ArrayRealVector(output);
